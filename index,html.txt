<script>
var firebaseConfig = {
  databaseURL: "https://save-the-hangman-default-rtdb.firebaseio.com"
};
firebase.initializeApp(firebaseConfig);
var db = firebase.database();
var roomCode = "";
var myRole = "";
var roomRef = null;
var gameState = null;
var CVS, CTX;
var hintUsed = false;
var KB_ROWS = [
  ["ض","ص","ث","ق","ف","غ","ع","ه","خ","ح","ج"],
  ["ش","س","ي","ب","ل","ا","ت","ن","م","ك"],
  ["ئ","ء","ؤ","ر","ى","ة","و","ز","ظ","ط","ذ","د"]
];
function showScreen(id){document.querySelectorAll(".screen").forEach(function(s){s.classList.remove("active");});document.getElementById(id).classList.add("active");}
function createRoom(){roomCode=Math.floor(1000+Math.random()*9000).toString();myRole="setter";roomRef=db.ref("rooms/"+roomCode);roomRef.set({status:"waiting_word",word:"",category:"",wtype:"",hint:"",guessed:[],wrong:0,gameOver:false,winner:""});document.getElementById("roomCodeDisplay").textContent=roomCode;showScreen("wordEntryScreen");}
function submitWord(){var w=document.getElementById("wordInput").value.trim();if(w.length<2){showToast("اكتب كلمة صحيحة! ✋");return;}roomRef.update({word:w,category:"🎯 كلمة مخصصة",wtype:"اختارها صديقك",hint:"فكّر كويس! 🤔",status:"waiting_guesser"});document.getElementById("wordInput").style.display="none";document.querySelector("#wordEntryScreen .big-btn").style.display="none";document.getElementById("waitingMsg").style.display="block";roomRef.on("value",function(snap){var data=snap.val();if(data&&data.status==="playing"){roomRef.off();startGame(data);}});}
function joinRoom(){var code=document.getElementById("joinCodeInput").value.trim();if(code.length!==4){showToast("الكود 4 أرقام! 🔢");return;}roomCode=code;myRole="guesser";roomRef=db.ref("rooms/"+roomCode);roomRef.once("value",function(snap){var data=snap.val();if(!data){showToast("غرفة غير موجودة! ❌");return;}if(data.status==="waiting_word"){showToast("الغرفة لسا ما اتجهزت ⏳");return;}if(data.status==="playing"||data.status==="done"){showToast("الجولة بدأت! ادخل غرفة ثانية");return;}roomRef.update({status:"playing"});startGame(data);});}
function startGame(data){gameState=data;hintUsed=false;showScreen("gameScreen");initCanvas();buildUI();var rb=document.getElementById("roleBadge");rb.textContent=myRole==="setter"?"👤 أنت: واضع الكلمة":"🎮 أنت: المخمّن";rb.className="role-badge "+(myRole==="setter"?"role-setter":"role-guesser");document.getElementById("roomMini").textContent="غرفة #"+roomCode;if(myRole==="setter"){document.getElementById("keyboard").style.display="none";document.getElementById("setterWatch").style.display="flex";document.getElementById("hintBtn").style.display="none";}else{document.getElementById("keyboard").style.display="flex";document.getElementById("setterWatch").style.display="none";buildKB();}roomRef.on("value",function(snap){var d=snap.val();if(!d)return;gameState=d;updateUI(d);if(d.gameOver&&d.status==="done"){roomRef.off();setTimeout(function(){openResult(d);},600);}});}
function initCanvas(){CVS=document.getElementById("gallows");var size=Math.min(window.innerHeight*0.22,window.innerWidth*0.38,130);CVS.width=Math.round(size*1.25);CVS.height=Math.round(size);CTX=CVS.getContext("2d");}
function buildUI(){document.getElementById("catEl").textContent=gameState.category||"";document.getElementById("typeEl").textContent="📌 "+(gameState.wtype||"");buildWord();updateLives(gameState.wrong||0);drawG(gameState.wrong||0);}
function updateUI(d){drawG(d.wrong||0);updateLives(d.wrong||0);buildWordFromData(d);updateTried(d);if(myRole==="setter")updateSetterView(d);if(myRole==="guesser")updateKBColors(d);}
function buildWord(){buildWordFromData(gameState);}
function buildWordFromData(d){var row=document.getElementById("wordEl");row.innerHTML="";var word=d.word||"";var guessed=d.guessed||[];word.split("").forEach(function(ch){var b=document.createElement("div");if(ch===" "){b.className="lb space";}else{b.className="lb";if(guessed.indexOf(ch)>=0||d.gameOver){b.textContent=ch;b.classList.add(guessed.indexOf(ch)>=0?"revealed":"wrong-reveal");}}row.appendChild(b);});}
function updateTried(d){var el=document.getElementById("triedEl");el.innerHTML="";var guessed=d.guessed||[];var word=d.word||"";guessed.forEach(function(l){var s=document.createElement("span");s.textContent=l+" ";s.className="tc "+(word.indexOf(l)>=0?"hit":"miss");el.appendChild(s);});}
function updateLives(wrong){var h="";var rem=6-wrong;for(var i=0;i<6;i++)h+=(i<rem?"❤️":"🖤");document.getElementById("livesEl").innerHTML=h;}
function updateSetterView(d){var el=document.getElementById("setterLetters");el.innerHTML="";var guessed=d.guessed||[];var word=d.word||"";guessed.forEach(function(l){var b=document.createElement("div");b.className="setter-letter "+(word.indexOf(l)>=0?"hit":"miss");b.textContent=l;el.appendChild(b);});document.getElementById("wrongCount").textContent="أخطاء: "+(d.wrong||0)+" / 6";var partial=word.split("").map(function(ch){if(ch===" ")return "  ";return guessed.indexOf(ch)>=0?ch:"_";}).join(" ");document.getElementById("setterWordReveal").textContent=partial;}
function buildKB(){var kb=document.getElementById("keyboard");kb.innerHTML="";KB_ROWS.forEach(function(row){var r=document.createElement("div");r.className="kb-row";row.forEach(function(L){var b=document.createElement("button");b.className="key";b.textContent=L;b.id="k_"+L;(function(letter){b.addEventListener("touchstart",function(e){e.preventDefault();doGuess(letter);},{passive:false});b.addEventListener("click",function(){doGuess(letter);});})(L);r.appendChild(b);});kb.appendChild(r);});}
function updateKBColors(d){var guessed=d.guessed||[];var word=d.word||"";guessed.forEach(function(l){var btn=document.getElementById("k_"+l);if(btn){btn.className="key used "+(word.indexOf(l)>=0?"hit":"miss");}});}
function doGuess(letter){if(myRole!=="guesser")return;var d=gameState;if(d.gameOver)return;var guessed=d.guessed?d.guessed.slice():[];if(guessed.indexOf(letter)>=0){showToast("جربته! ↩️");return;}guessed.push(letter);var word=d.word||"";var wrong=d.wrong||0;var isHit=word.indexOf(letter)>=0;if(!isHit)wrong++;var allFound=true;word.split("").forEach(function(ch){if(ch!==" "&&guessed.indexOf(ch)<0)allFound=false;});var gameOver=allFound||wrong>=6;var winner=allFound?"guesser":(wrong>=6?"setter":"");roomRef.update({guessed:guessed,wrong:wrong,gameOver:gameOver,winner:winner,status:gameOver?"done":"playing"});}
function useHint(){if(hintUsed||myRole!=="guesser")return;hintUsed=true;document.getElementById("hintBtn").classList.add("used");document.getElementById("hintText").textContent="💡 "+(gameState.hint||"");showToast("تلميح مكشوف! 💡");}
function openResult(d){var guesserWon=d.winner==="guesser";var iWon=(myRole==="guesser"&&guesserWon)||(myRole==="setter"&&!guesserWon);document.getElementById("resEm").textContent=iWon?"🎉":"😔";document.getElementById("resTitle").textContent=iWon?"فزت! 🏆":"خسرت! 💀";document.getElementById("resSub").textContent=guesserWon?"المخمّن اكتشف الكلمة!":"المخمّن ما قدر يكتشفها!";document.getElementById("resWord").textContent=d.word;if(iWon)confetti();document.getElementById("resOv").classList.add("show");}
